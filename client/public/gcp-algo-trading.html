Sample Python GCP Trading Code

Below is a sample Python script for a momentum-based trading bot on GCP Compute Engine, using `ccxt` for Binance and a 1:3 risk-reward ratio: 


import ccxt
import time
import numpy as np
import os
from datetime import datetime
from google.cloud import secretmanager

# Access Binance API keys from GCP Secret Manager
client = secretmanager.SecretManagerServiceClient()
project_id = "your-gcp-project-id"
api_key_secret = client.access_secret_version(name=f"projects/{project_id}/secrets/binance-api-key/versions/latest")
api_secret_secret = client.access_secret_version(name=f"projects/{project_id}/secrets/binance-secret/versions/latest")
BINANCE_API_KEY = api_key_secret.payload.data.decode("UTF-8")
BINANCE_SECRET = api_secret_secret.payload.data.decode("UTF-8")

# Initialize Binance client
binance = ccxt.binance({
    'apiKey': BINANCE_API_KEY,
    'secret': BINANCE_SECRET,
    'enableRateLimit': True
})

SYMBOL = 'BTC/USDT'
TIMEFRAME = '1m'  # 1-minute candles
POSITION_SIZE = 0.001  # Trade 0.001 BTC per order
STOP_LOSS = 0.0075  # 0.75% stop-loss
TAKE_PROFIT = 0.0225  # 2.25% take-profit (1:3 risk-reward)

def calculate_rsi(data, period=14):
    deltas = np.diff(data)
    gains = deltas[deltas > 0]
    losses = -deltas[deltas < 0]
    avg_gain = np.mean(gains[-period:]) if len(gains) >= period else 0
    avg_loss = np.mean(losses[-period:]) if len(losses) >= period else 0
    rs = avg_gain / avg_loss if avg_loss != 0 else 0
    return 100 - (100 / (1 + rs))

def main():
    print(f"Starting trading bot for {SYMBOL} at {datetime.now()}")
    while True:
        try:
            # Fetch 1-minute OHLCV data
            ohlcv = binance.fetch_ohlcv(SYMBOL, TIMEFRAME, limit=50)
            closes = [c[4] for c in ohlcv]

            # Calculate RSI
            rsi = calculate_rsi(closes)
            current_price = closes[-1]

            # Check balance
            balance = binance.fetch_balance()['USDT']['free']
            if balance < 10:  # Ensure minimum trade size
                print(f"Insufficient balance: {balance}")
                time.sleep(60)
                continue

            # Trading logic: RSI-based momentum
            if rsi > 70:  # Overbought, potential sell
                print(f"Sell signal at {current_price} (RSI: {rsi:.2f})")
                order = binance.create_market_sell_order(SYMBOL, POSITION_SIZE)
                print(f"Sell order placed: {order['id']}")
            elif rsi < 30:  # Oversold, potential buy
                print(f"Buy signal at {current_price} (RSI: {rsi:.2f})")
                order = binance.create_market_buy_order(SYMBOL, POSITION_SIZE)
                entry_price = current_price
                print(f"Buy order placed: {order['id']}")

                # Monitor for stop-loss or take-profit
                while True:
                    price = binance.fetch_ticker(SYMBOL)['last']
                    if price <= entry_price * (1 - STOP_LOSS):
                        binance.create_market_sell_order(SYMBOL, POSITION_SIZE)
                        print(f"Stop-loss hit at {price}")
                        break
                    if price >= entry_price * (1 + TAKE_PROFIT):
                        binance.create_market_sell_order(SYMBOL, POSITION_SIZE)
                        print(f"Take-profit hit at {price}")
                        break
                    time.sleep(5)
            else:
                print(f"No signal, holding... (RSI: {rsi:.2f})")

            time.sleep(60)  # Check every minute

        except Exception as e:
            print(f"Error: {e}")
            time.sleep(60)

if __name__ == "__main__":
    main()
